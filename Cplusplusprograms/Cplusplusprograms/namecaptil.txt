# Load the address of the input string into $a0
la $a0, str_input

# Load a pointer to the beginning of the string into $t0
move $t0, $a0

# Loop through the string until we reach the end
loop:
    lb $t1, ($t0)       # Load the next character into $t1
    beqz $t1, exit      # If the character is null, exit the loop
    bltu $t1, 97, next  # If the character is not a lowercase letter, skip to next character
    
    addi $t1, $t1, -32  # Convert the lowercase letter to uppercase by subtracting 32 from its ASCII code
    
    sb $t1, ($t0)       # Store the new uppercase letter back into the string
    
next:
    addi $t0, $t0, 1    # Move to the next character in the string
    j loop              # Repeat the loop

# Exit the program
exit:
    jr $ra
